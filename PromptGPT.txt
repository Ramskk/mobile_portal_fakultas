class ScraperService {
  static const String beritaUrl = 'https://fst.umsida.ac.id/';
  
  Future<List<Map<String, String>>> fetchBerita() async {
    final response = await http.get(Uri.parse(beritaUrl));
    if (response.statusCode != 200) return [];

    final document = parser.parse(response.body);
    final elements = document.querySelectorAll('.post-item'); // ubah sesuai selector HTML
    
    return elements.map((element) {

      final link = element.querySelector('a')?.attributes['href'] ?? '';

    }).toList();

services/scraper_service.dart
import 'package:http/http.dart' as http;
import 'package:html/parser.dart' as parser;
import 'package:html/dom.dart';

class ScraperService {
  static const String beritaUrl = 'https://fst.umsida.ac.id/category/berita/';

  Future<List<Map<String, String>>> fetchBerita() async {
    final response = await http.get(Uri.parse(beritaUrl));
    if (response.statusCode != 200) {
      return [];
    }

    final document = parser.parse(response.body);
    // Selector: mencari elemen artikel berita
    final List<Element> items = document.querySelectorAll('article.post');

    return items.map((element) {
      final titleElem = element.querySelector('h2.entry-title a');
      final imgElem = element.querySelector('img');
      final link = titleElem?.attributes['href'] ?? '';
      final title = titleElem?.text.trim() ?? '';
      final img = imgElem?.attributes['src'] ?? '';

      return {
        'title': title,
        'link': link,
        'img': img,
      };
    }).toList();
  }
}

import 'package:flutter/material.dart';
import '../services/scraper_service.dart';
import '../widgets/berita_card.dart';

class BeritaPage extends StatefulWidget {
  @override
  State<BeritaPage> createState() => _BeritaPageState();
}

class _BeritaPageState extends State<BeritaPage> {
  final ScraperService scraper = ScraperService();
  List<Map<String, String>> beritaList = [];
  bool isLoading = true;

  @override
  void initState() {
    super.initState();
    loadBerita();
  }

  Future<void> loadBerita() async {
    final data = await scraper.fetchBerita();
    setState(() {
      beritaList = data;
      isLoading = false;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Berita FST UMSIDA'),
      ),
      body: isLoading
        ? Center(child: CircularProgressIndicator())
        : ListView.builder(
            itemCount: beritaList.length,
            itemBuilder: (context, index) {
              final item = beritaList[index];
              return BeritaCard(
                title: item['title'] ?? '',
                imageUrl: item['img'] ?? '',
                link: item['link'] ?? '',
              );
            },
          ),
    );
  }
}

import 'package:http/http.dart' as http;

class ApiService {
  static const String baseUrl = 'https://fst.umsida.ac.id/wp-json/wp/v2/posts?_embed&per_page=10';

  Future<List<Map<String, dynamic>>> fetchPosts() async {
    final response = await http.get(Uri.parse(baseUrl));
    if (response.statusCode != 200) return [];

    final List data = jsonDecode(response.body);
    return data.map((post) {
      final embedded = post['_embedded'];
      String imageUrl = '';
      if (embedded != null &&
          embedded['wp:featuredmedia'] != null &&
          embedded['wp:featuredmedia'].isNotEmpty) {
        imageUrl = embedded['wp:featuredmedia'][0]['source_url'] ?? '';
      }

      return {
        'title': post['title']['rendered'],
        'excerpt': post['excerpt']['rendered'],
        'link': post['link'],
        'image': imageUrl,
        'date': post['date'],
      };
    }).toList();
  }
}

class ApiService {
  static const String baseUrl =
      'https://fst.umsida.ac.id/wp-json/wp/v2/posts?_embed&per_page=10';

  Future<List<Map<String, dynamic>>> fetchPosts() async {
    try {
      final response = await http.get(Uri.parse(baseUrl));
      if (response.statusCode != 200) return [];

      final List data = jsonDecode(response.body);

      return data.map((post) {
        final embedded = post['_embedded'];
        String imageUrl = '';
        if (embedded != null &&
            embedded['wp:featuredmedia'] != null &&
            embedded['wp:featuredmedia'].isNotEmpty) {
          imageUrl = embedded['wp:featuredmedia'][0]['source_url'] ?? '';
        }

        return {
          'title': post['title']['rendered'],
          'excerpt': post['excerpt']['rendered'],
          'link': post['link'],
          'image': imageUrl,
          'date': post['date'],
        };
      }).toList();
    } catch (e) {
      print('Error fetching posts: $e');
      return [];
    }
  }
}
